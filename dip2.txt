  {

  },
  {
    question: "The Laplacian operator is used in image processing for:",
    options: ["A) Image sharpening", "B) Image smoothing", "C) Color conversion", "D) Image compression"],

  },
  {
    question: "Histogram equalization is primarily used to:",

    answer: "B) Enhance contrast"
  },
  {
    question: "Morphological operations in digital image processing are mainly applied to:",

  },
  {
    question: "Edge detection in an image helps to:",

  {

  }

  let current = 0;
  let score = 0;

  function loadQuiz() {
    const quiz = document.getElementById("quiz");
    quiz.innerHTML = "";

    const q = quizData[current];
    quiz.innerHTML = `<div class="question">${current + 1}. ${q.question}</div>`;

    q.options.forEach(opt => {
      quiz.innerHTML += `
        <label>
          <input type="radio" name="answer" value="${opt}"> ${opt}
        </label>
      `;
    });
  }

  function nextQuestion() {
    const selected = document.querySelector('input[name="answer"]:checked');

    if (!selected) {
      alert("Answer select pannunga!");
      return;
    }

    if (selected.value === quizData[current].answer) {
      score++;
    }

    current++;

    if (current < quizData.length) {
      loadQuiz();
    } else {
      document.getElementById("quiz").innerHTML = "";
      document.getElementById("result").innerHTML =
        `You scored ${score} out of ${quizData.length} ??`;
    }
  }

  loadQuiz();
